import java.util.Arrays;
import java.util.Scanner;
import java.util.Random;

public class hangman {
    private static String generatePhrase(){
        String[] phrases = {"armadillo", "banana", "chocolate"};
        Random rand = new Random();
        int num = rand.nextInt(3);
        return phrases[num];
    }
    
    private static String checkForMatch(String letter, String word){
        if(word[0].equals(letter)){
            return "match";
        } else {
            return "inequal";
        }
    }
    
    public static void main(String[] args){
        String guessWord = generatePhrase();
        String userAnswer = "";
        
        int wrongGuesses = 0;
        while(wrongGuesses < 10){
            //User Input Letter
            Scanner myObj = new Scanner(System.in);
            System.out.println("Enter Letter");
            String userLetter = myObj.nextLine();
            if(checkForMatch(userLetter, guessWord) == "match"){
                //update user answer
            } else {
                wrongGuesses++;
            }
        }
        
        if(userAnswer.equals(guessWord)){
            System.out.println("You Win!");
        } else {
            System.out.println("You Lose!");
        }
    }
}
